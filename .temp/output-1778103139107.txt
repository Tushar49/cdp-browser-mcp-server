                  if(payload_data.status==200){
                    $('#payment_status_data').text(payload_data.message);

                    $('#payment_status_data').text(payload_data.message);
                  }
                }
                else{
                   $('#payment_status_data').text('Error while fetching payment status, please try after some time');
                }
            },
            error:function (response) {
               $(".overlay").hide();
               if (response){
                 let XSSI_PREFIX =  ")]}'"
                 response_text = response.responseText
                 let payload_data = JSON.parse(response_text.replace(XSSI_PREFIX, ''))
                 $('#payment_status_data').text(payload_data.message);
               }
            }
        });
    }

    function downloadHallTicket(){
        $.ajax({
            type: "GET",
            url: 'download_hall_ticket',
            dataType: 'text',
            success: function(response){
                if(response){
                    let XSSI_PREFIX = ")]}'"
                    let payload_data = JSON.parse(response.replace(XSSI_PREFIX, ''));

                    if(payload_data.payload){
                        let payload = JSON.parse(payload_data.payload);
                        if(payload.status == 'success' && payload.download_url){
                            window.open(payload.download_url, '_blank');
                        } else {
                            alert(payload_data.message || 'Unable to download hall ticket.');
                        }
                    } else {
                        alert(payload_data.message || 'Unable to download hall ticket.');
                    }
                } else {
                    alert('Error while downloading hall ticket, please try again later.');
                }
            },
            error: function (response) {
                if (response && response.responseText){
                    let XSSI_PREFIX = ")]}'"
                    try {
                        let payload_data = JSON.parse(response.responseText.replace(XSSI_PREFIX, ''));
                        alert(payload_data.message || 'Unable to download hall ticket.');
                    } catch (e) {
                        alert('Error while downloading hall ticket, please try again later.');
                    }
                } else {
                    alert('Network error. Please check your connection and try again.');
                }
            }
        });
    }
